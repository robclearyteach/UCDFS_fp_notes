##
# Notice above: when on '*main' 
#	git log 
#	shows no 'featureZ' log record: why?
#
#	 featureZ commits are not part of main history.
#
#	* git log shows the history of the current branch only
#	* main and featureZ are on different branch pointers
#	* featureZ commit is only reachable from featureZ, not from main
#

##
# Also notice: 
#  the initial commit doesn't have (main) 
#	along-side it 
commit cedd4ab35d4216c0358d77f66dd60cb3419b087b              # NO '(main) marker
Author...
Date...
    initial commit:
#
# because 'main' is just a pointer and while it 
# *was* pointing to the initial commit, after the 
# 'first line' add/commit it then moved to the 
# next (that) commit.
